15.         //在要表示对象的属性前加上this.来修饰 ，因为this代表的就是你创建的那个对象
16.         this.name = name;
17.         this.age = age;
18.         this.height = height;
19.     }
20.     public Person(String a,int b){
21.         name = a;
22.         age = b;
23.     }
